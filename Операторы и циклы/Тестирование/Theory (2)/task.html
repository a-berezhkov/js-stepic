Импорт функций 

Давайте теперь создадим отдельный файл "index.js" с набором функций экспортируем их.

/**
 * Фцнкция сложения
 * @param {*} a 
 * @param {*} b 
 * @returns 
 */
function sum(a,b){
    return a+b;
}
/**
 * Функция вычитания
 * @param {*} a 
 * @param {*} b 
 * @returns 
 */
function minus(a,b){
    return a - b
}
exports.sum = sum
exports.minus = minus

В примере используются  exports для передачи методов в другой файл. Добавим теперь несколько методов для проверки в "test.js".

const functions = require('./index.js');
const assert = require('assert');
 
  describe('#sum', function () {
    it('сложение 2 + 3' , function () {
      assert.equal(functions.sum(2,3), 5);
    });
    it('сложение 100  и -99' , function () {
      assert.equal(functions.sum(100,-99), 1);
    });
    it('сложение -11  и -22' , function () {
      assert.equal(functions.sum(-11,-22), -33);
    });
    it('сложение 0.11  и 0.22' , function () {
      assert.equal(functions.sum(0.11,0.22), 0.33);
    });
  });

   
  describe('#minus', function () {
    it('вычитание 2 - 3' , function () {
      assert.equal(functions.minus(2,3), -1);
    });
    it('вычитание 100  и -99' , function () {
      assert.equal(functions.minus(100,-99), 199);
    });
    it('вычитание -11  и -22' , function () {
      assert.equal(functions.minus(-11,-22), 11);
    });
    it('вычитание 0.11  и 0.22' , function () {
      assert.equal(functions.minus(0.11,0.22), -0.11);
    });
  });

Аналогичным образом могут импортированы классы, объекты, переменные.
